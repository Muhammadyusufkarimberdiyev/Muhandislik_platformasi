// LED Blink va Fade (Arduino)
// Ulanish: LED (+) -> D9 (PWM), LED (-) -> 220Œ© -> GND
const int LED = 9;

void setup(){
  pinMode(LED, OUTPUT);
}

void loop(){
  // 1) Blink
  digitalWrite(LED, HIGH);
  delay(500);
  digitalWrite(LED, LOW);
  delay(500);

  // 2) Fade (0..255..0)
  for (int b=0; b<=255; b+=5){ analogWrite(LED, b); delay(15); }
  for (int b=255; b>=0; b-=5){ analogWrite(LED, b); delay(15); }
}

// LED Blink - Asosiy dastur
const int ledPin = 13;

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
  Serial.println("LED Blink tajribasi boshlandi!");
}

void loop() {
  digitalWrite(ledPin, HIGH);
  Serial.println("LED ON");
  delay(1000);
  
  digitalWrite(ledPin, LOW);
  Serial.println("LED OFF");
  delay(1000);
}
