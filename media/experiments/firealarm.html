// 360¬∞ Fire Alarm System
#define BUZZER_PIN 9
#define LED_PIN 13
#define TRANSISTOR_PIN 8

// IR Sensor pins
int sensorPins[] = {2, 3, 4, 5, 6, 7, 8, 12};
bool fireDetected = false;

void setup() {
  Serial.begin(9600);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(LED_PIN, OUTPUT);
  pinMode(TRANSISTOR_PIN, OUTPUT);
  
  for(int i = 0; i < 8; i++) {
    pinMode(sensorPins[i], INPUT);
  }
}

void loop() {
  fireDetected = false;
  
  // Check all sensors
  for(int i = 0; i < 8; i++) {
    if(digitalRead(sensorPins[i]) == LOW) {
      fireDetected = true;
      Serial.print("Fire detected on sensor: ");
      Serial.println(i+1);
      break;
    }
  }
  
  if(fireDetected) {
    activateAlarm();
  } else {
    deactivateAlarm();
  }
  
  delay(100);
}

void activateAlarm() {
  digitalWrite(LED_PIN, HIGH);
  digitalWrite(TRANSISTOR_PIN, HIGH);
  
  // Buzzer sound pattern
  for(int i = 0; i < 3; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    delay(200);
    digitalWrite(BUZZER_PIN, LOW);
    delay(100);
  }
}

void deactivateAlarm() {
  digitalWrite(LED_PIN, LOW);
  digitalWrite(TRANSISTOR_PIN, LOW);
  digitalWrite(BUZZER_PIN, LOW);
}
