// Sumo Car â€“ HC-SR04 + 2Ã— IR (ring cheti) + L293D/L298N
// Pinlar (o'zingiz moslashtirishingiz mumkin):
// Ultrasonic: TRIG D6, ECHO D7
// IR edge sensori: LEFT A0, RIGHT A1 (digital sifatida o'qiyapmiz)
// Motor driver (L293D/L298N):
//   ENA D3 (PWM), IN1 D8, IN2 D9   -> Chap motor
//   ENB D11(PWM), IN3 D12, IN4 D13 -> O'ng motor

const int TRIG = 6;
const int ECHO = 7;

const int IR_L = A0;
const int IR_R = A1;

const int ENA = 3, IN1 = 8,  IN2 = 9;
const int ENB = 11,IN3 = 12, IN4 = 13;

int SPD_FAST = 255;
int SPD_SLOW = 170;
int EDGE_WHITE = 1; // ko'p IR modullarda oq/chetda 1 chiqadi, lozim bo'lsa teskari qiling

long distanceCM(){
  digitalWrite(TRIG, LOW); delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  long us = pulseIn(ECHO, HIGH, 30000UL); // 30ms = ~5m
  if(us==0) return 999;
  return us/58;
}

void motors(int lPWM, bool lForward, int rPWM, bool rForward){
  // Chap
  digitalWrite(IN1, lForward ? HIGH : LOW);
  digitalWrite(IN2, lForward ? LOW  : HIGH);
  analogWrite(ENA, constrain(lPWM,0,255));
  // O'ng
  digitalWrite(IN3, rForward ? HIGH : LOW);
  digitalWrite(IN4, rForward ? LOW  : HIGH);
  analogWrite(ENB, constrain(rPWM,0,255));
}

void fwd(int sp){ motors(sp,true,sp,true); }
void back(int sp){ motors(sp,false,sp,false); }
void spinLeft(int sp){ motors(sp,false,sp,true); }
void spinRight(int sp){ motors(sp,true,sp,false); }
void stopM(){ motors(0,true,0,true); }

void setup(){
  pinMode(TRIG, OUTPUT); pinMode(ECHO, INPUT);
  pinMode(IR_L, INPUT);  pinMode(IR_R, INPUT);
  pinMode(ENA, OUTPUT);  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(ENB, OUTPUT);  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  stopM();
}

void loop(){
  // 1) Ring chetini tekshirish (oq chiziqdan qaytish)
  int leftEdge  = digitalRead(IR_L);
  int rightEdge = digitalRead(IR_R);

  if(leftEdge==EDGE_WHITE && rightEdge!=EDGE_WHITE){
    // Chap tomonda chet: o'ngga qoch
    back(SPD_SLOW); delay(180);
    spinRight(SPD_FAST); delay(200);
    return;
  }
  if(rightEdge==EDGE_WHITE && leftEdge!=EDGE_WHITE){
    // O'ng tomonda chet: chapga qoch
    back(SPD_SLOW); delay(180);
    spinLeft(SPD_FAST); delay(200);
    return;
  }
  if(leftEdge==EDGE_WHITE && rightEdge==EDGE_WHITE){
    // Ikkala tomonda ham oq â€“ orqaga va buril
    back(SPD_SLOW); delay(250);
    spinLeft(SPD_FAST); delay(250);
    return;
  }

  // 2) Raqibni qidirish / hujum
  long d = distanceCM();
  if(d < 30){             // raqib yaqin bo'lsa â€“ hujum
    fwd(SPD_FAST);
    delay(60);
  }else{                   // topilmasa â€“ aylanish orqali qidirish
    spinLeft(SPD_SLOW);
  }
}
