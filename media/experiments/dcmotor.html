// DC Motor Forward/Reverse Control
const int forwardButton = 2;
const int reverseButton = 3;
const int motorPin1 = 4;
const int motorPin2 = 5;

void setup() {
  pinMode(forwardButton, INPUT_PULLUP);
  pinMode(reverseButton, INPUT_PULLUP);
  pinMode(motorPin1, OUTPUT);
  pinMode(motorPin2, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  if (digitalRead(forwardButton) == LOW) {
    // Forward direction
    digitalWrite(motorPin1, HIGH);
    digitalWrite(motorPin2, LOW);
    Serial.println("Motor: Forward");
  }
  else if (digitalRead(reverseButton) == LOW) {
    // Reverse direction
    digitalWrite(motorPin1, LOW);
    digitalWrite(motorPin2, HIGH);
    Serial.println("Motor: Reverse");
  }
  else {
    // Stop motor
    digitalWrite(motorPin1, LOW);
    digitalWrite(motorPin2, LOW);
    Serial.println("Motor: Stop");
  }
  delay(50);
}
                
import RPi.GPIO as GPIO
import time

# GPIO pin setup
forward_button = 18
reverse_button = 19
motor_pin1 = 20
motor_pin2 = 21

GPIO.setmode(GPIO.BCM)
GPIO.setup([forward_button, reverse_button], GPIO.IN, pull_up_down=GPIO.PUD_UP)
GPIO.setup([motor_pin1, motor_pin2], GPIO.OUT)

try:
    while True:
        if not GPIO.input(forward_button):
            # Forward direction
            GPIO.output(motor_pin1, GPIO.HIGH)
            GPIO.output(motor_pin2, GPIO.LOW)
            print("Motor: Forward")
        elif not GPIO.input(reverse_button):
            # Reverse direction
            GPIO.output(motor_pin1, GPIO.LOW)
            GPIO.output(motor_pin2, GPIO.HIGH)
            print("Motor: Reverse")
        else:
            # Stop motor
            GPIO.output(motor_pin1, GPIO.LOW)
            GPIO.output(motor_pin2, GPIO.LOW)
            print("Motor: Stop")
        
        time.sleep(0.05)

except KeyboardInterrupt:
    GPIO.cleanup()
