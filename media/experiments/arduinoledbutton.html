// LED Blink va Fade (Arduino)
// Ulanish: LED (+) -> D9 (PWM), LED (-) -> 220Î© -> GND
const int LED = 9;

void setup(){
  pinMode(LED, OUTPUT);
}

void loop(){
  // 1) Blink
  digitalWrite(LED, HIGH);
  delay(500);
  digitalWrite(LED, LOW);
  delay(500);

  // 2) Fade (0..255..0)
  for (int b=0; b<=255; b+=5){ analogWrite(LED, b); delay(15); }
  for (int b=255; b>=0; b-=5){ analogWrite(LED, b); delay(15); }
}
