DHT11 + LCD Loyihasi

Bu loyiha DHT11 sensor yordamida harorat va namlikni o'lchab, LCD displeyda ko'rsatadi. Arduino UNO asosida ishlaydi va real vaqtda ma'lumotlarni yangilaydi.

Kod

#include <DHT.h>
#include <LiquidCrystal.h>

#define DHTPIN 2
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

void setup() {
  Serial.begin(9600);
  dht.begin();
  lcd.begin(16, 2);
  lcd.print("DHT11 Sensor");
  delay(2000);
  lcd.clear();
}

void loop() {
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature();

  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("DHT11 o'qishda xatolik!");
    lcd.setCursor(0, 0);
    lcd.print("Sensor xatolik!");
    return;
  }

  // LCD ga chiqarish
  lcd.setCursor(0, 0);
  lcd.print("Temp: ");
  lcd.print(temperature);
  lcd.print("C");
  
  lcd.setCursor(0, 1);
  lcd.print("Humidity: ");
  lcd.print(humidity);
  lcd.print("%");

  // Serial monitor
  Serial.print("Harorat: ");
  Serial.print(temperature);
  Serial.print("¬∞C, Namlik: ");
  Serial.print(humidity);
  Serial.println("%");

  delay(2000);
}
