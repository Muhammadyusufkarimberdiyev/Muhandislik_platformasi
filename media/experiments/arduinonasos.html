void setup() {
  Serial.begin(9600);
  pinMode(13, OUTPUT);
}

void loop() {
  if (Serial.available() > 0) {
    data = Serial.readString();
    
    if (data == "ON") {
      digitalWrite(13, HIGH);
    }
    if (data == "OFF") {
      digitalWrite(13, LOW);
    }
  }
}
