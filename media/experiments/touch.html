üíª Arduino Kodi

// Touch Switch with BC547 Transistor
const int touchPin = A0;  // Analog pin for touch sensing
const int ledPin = 13;    // Digital pin for LED
int threshold = 512;      // Touch threshold value

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
  Serial.println("Touch Switch Ready!");
}

void loop() {
  int touchValue = analogRead(touchPin);
  
  if (touchValue > threshold) {
    digitalWrite(ledPin, HIGH);  // LED ON
    Serial.println("Touch detected - LED ON");
  } else {
    digitalWrite(ledPin, LOW);   // LED OFF
    Serial.println("No touch - LED OFF");
  }
  
  Serial.print("Touch value: ");
  Serial.println(touchValue);
  delay(100);
}
