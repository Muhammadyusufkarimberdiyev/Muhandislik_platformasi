// Obstacle-avoiding Vacuum Robot (HC-SR04 + Servo scan + L298N)
// Pinlar:
#define TRIG 9
#define ECHO 8
#define SERVO_PIN 10      // ixtiyoriy (yo'q bo'lsa, CENTER_ANGLE 90 ga fiks qilib ishlang)
#define ENA 3             // PWM chap motor
#define IN1 4
#define IN2 5
#define IN3 6
#define IN4 7
#define ENB 11            // PWM o'ng motor
// Ixtiyoriy: fan/so'ruvchi motorni relay orqali yoqish
#define FAN_RELAY 12      // HIGH=ON (o'zingizga moslang)

#include 
Servo head;

long measureCM(){
  digitalWrite(TRIG, LOW); delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  long dur = pulseIn(ECHO, HIGH, 25000); // ~4m
  if(dur==0) return 500;
  return dur/58;
}

void motors(int ls, int rs){ // -255..255
  ls = constrain(ls, -255, 255);
  rs = constrain(rs, -255, 255);
  // chap
  if(ls >= 0){ digitalWrite(IN1,HIGH); digitalWrite(IN2,LOW);  analogWrite(ENA, ls); }
  else       { digitalWrite(IN1,LOW);  digitalWrite(IN2,HIGH); analogWrite(ENA, -ls); }
  // o'ng
  if(rs >= 0){ digitalWrite(IN3,HIGH); digitalWrite(IN4,LOW);  analogWrite(ENB, rs); }
  else       { digitalWrite(IN3,LOW);  digitalWrite(IN4,HIGH); analogWrite(ENB, -rs); }
}

void stopMotors(){ analogWrite(ENA,0); analogWrite(ENB,0); digitalWrite(IN1,LOW); digitalWrite(IN2,LOW); digitalWrite(IN3,LOW); digitalWrite(IN4,LOW); }

int scanBestDir(){
  // servo bo'lsa: 30Â°â€“150Â° oralig'ida skan, eng uzoq yo'nalishni qaytaradi
  int bestAngle = 90, bestDist = -1;
  for(int a=30;a<=150;a+=30){
    head.write(a); delay(150);
    int d = measureCM();
    if(d>bestDist){ bestDist=d; bestAngle=a; }
  }
  head.write(90);
  return bestAngle;
}

void setup(){
  pinMode(TRIG, OUTPUT); pinMode(ECHO, INPUT);
  pinMode(ENA, OUTPUT); pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(ENB, OUTPUT); pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  pinMode(FAN_RELAY, OUTPUT); digitalWrite(FAN_RELAY, HIGH); // fan ON
  head.attach(SERVO_PIN); head.write(90);
  delay(400);
}

void loop(){
  int d = measureCM();
  if(d > 25){ // yo'l ochiq
    motors(180, 180);
  }else{      // to'siq
    stopMotors();
    int a = scanBestDir();      // servo mavjud bo'lmasa: a= (random(0,2)?45:135);
    if(a < 90){ // chap
      motors(-160,160); delay(350);
    }else{      // o'ng
      motors(160,-160); delay(350);
    }
  }
  delay(20);
}
