Arduino Kodi

// LED Yashnash Tajribasi
// Uchta LED navbat bilan yonadi va o'chadi

// LED pinlarini belgilash
int redLED = 8;     // Qizil LED 8-pinga ulangan
int yellowLED = 9;  // Sariq LED 9-pinga ulangan  
int greenLED = 10;  // Yashil LED 10-pinga ulangan

void setup() {
  // LED pinlarini chiqish rejimida sozlash
  pinMode(redLED, OUTPUT);
  pinMode(yellowLED, OUTPUT);
  pinMode(greenLED, OUTPUT);
  
  // Serial moniterni boshlash
  Serial.begin(9600);
  Serial.println("LED Yashnash Tajribasi Boshlandi!");
}

void loop() {
  // Qizil LED ni yoqish
  digitalWrite(redLED, HIGH);
  Serial.println("Qizil LED yondi");
  delay(500);  // 0.5 soniya kutish
  
  digitalWrite(redLED, LOW);
  Serial.println("Qizil LED o'chdi");
  delay(100);
  
  // Sariq LED ni yoqish
  digitalWrite(yellowLED, HIGH);
  Serial.println("Sariq LED yondi");
  delay(500);
  
  digitalWrite(yellowLED, LOW);
  Serial.println("Sariq LED o'chdi");
  delay(100);
  
  // Yashil LED ni yoqish
  digitalWrite(greenLED, HIGH);
  Serial.println("Yashil LED yondi");
  delay(500);
  
  digitalWrite(greenLED, LOW);
  Serial.println("Yashil LED o'chdi");
  delay(100);
  
  // Barcha LEDlarni bir vaqtda yoqish
  digitalWrite(redLED, HIGH);
  digitalWrite(yellowLED, HIGH);
  digitalWrite(greenLED, HIGH);
  Serial.println("Barcha LEDlar yondi");
  delay(1000);  // 1 soniya
  
  // Barcha LEDlarni o'chirish
  digitalWrite(redLED, LOW);
  digitalWrite(yellowLED, LOW);
  digitalWrite(greenLED, LOW);  
  Serial.println("Barcha LEDlar o'chdi");
  delay(500);
}
