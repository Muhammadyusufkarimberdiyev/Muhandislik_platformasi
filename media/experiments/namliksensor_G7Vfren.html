üíª Kod

#include 

#define DHTPIN 2       // DATA simi ulangan pin (rasmdagi sariq sim)
#define DHTTYPE DHT11  // Agar DHT22 bo'lsa -> DHT22 deb yozing

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();

  Serial.println("DHT11 sensor ishga tushdi...");
}

void loop() {
  float humidity = dht.readHumidity();     // Namlik
  float temperature = dht.readTemperature(); // Harorat (¬∞C)

  // Xatolikni tekshirish
  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("Sensorni o'qib bo'lmadi! Tekshiring.");
    delay(1000);
    return;
  }

  Serial.print("Harorat: ");
  Serial.print(temperature);
  Serial.print(" ¬∞C   |   Namlik: ");
  Serial.print(humidity);
  Serial.println(" %");

  delay(2000); // 2 sekundda bir o‚Äòqish
}
