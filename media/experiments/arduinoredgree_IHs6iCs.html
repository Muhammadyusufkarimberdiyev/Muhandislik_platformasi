int redLED = 12;     // Qizil LED

void setup() {
  pinMode(redLED, OUTPUT);     // LED pinlarini chiqish sifatida belgilash
  pinMode(greenLED, OUTPUT);

void loop() {
  digitalWrite(redLED, HIGH);   // Qizil LED yoqiladi
  digitalWrite(greenLED, LOW);
  delay(500);

  digitalWrite(redLED, LOW);    // Yashil LED yoqiladi
  digitalWrite(greenLED, HIGH);
  delay(500);
}
