// Obstacle Avoidance Car (HC-SR04 + L298N)
// Ulanish:
// HC-SR04: TRIG->D9, ECHO->D8
// L298N: IN1->D2, IN2->D3 (chap motor)
//        IN3->D4, IN4->D5 (o'ng motor)
// ENA/ENB -> 5V (yoki PWM pinlar orqali tezlik)
// Vcc (motor) -> akkumulyator, 5V -> Arduino (ehtiyot bo'ling)

#define TRIG 9
#define ECHO 8

#define IN1 2
#define IN2 3
#define IN3 4
#define IN4 5

const int SAFE_CM = 20;   // To'siqka qadar minimal masofa (sm)
const int TURN_MS = 400;  // Burilish vaqti (ms) â€” tajribada moslang
const int BACK_MS = 250;  // Orqaga yurish vaqti

long measureCM(){
  digitalWrite(TRIG, LOW); delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  long duration = pulseIn(ECHO, HIGH, 25000UL); // 25ms timeout (~4m)
  if (duration == 0) return 999;               // timeout â€” juda uzoq deb qabul qilamiz
  long cm = duration / 58;                     // ~ us->sm
  return cm;
}

void motorsStop(){
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

void motorsForward(){
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void motorsBackward(){
  digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
}

void motorsLeft(){
  // chap motor orqaga, o'ng motor oldinga â€” joyida chapga burilish
  digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void motorsRight(){
  // chap motor oldinga, o'ng motor orqaga â€” joyida o'ngga burilish
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);  digitalWrite(IN4, HIGH);
}

void setup(){
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);

  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);

  motorsStop();
}

void loop(){
  long d = measureCM();

  if (d <= SAFE_CM){
    motorsStop();
    delay(80);

    // Orqaga biroz
    motorsBackward(); delay(BACK_MS);
    motorsStop();     delay(60);

    // Tasodifiy tomonga burilish (oddiy)
    if (millis() % 2 == 0){ motorsLeft();  } else { motorsRight(); }
    delay(TURN_MS);

    motorsStop();
    delay(60);
  } else {
    motorsForward();
  }

  delay(20);
}
