#include SPI.h>
#include nRF24L01.h>
#include RF24.h>

RF24 radio(9,10);
const byte address[6] = "00001";

void setup() {
  radio.begin();
  radio.openWritingPipe(address);
  radio.setPALevel(RF24_PA_MIN);
}

void loop() {
  int xValue = analogRead(A0);
  radio.write(&xValue, sizeof(xValue));
}
Receiver kodi
#include SPI.h>
#include nRF24L01.h>
#include RF24.h>
#include Servo.h>

RF24 radio(9,10);
Servo myservo;
int xValue;

const byte address[6] = "00001";

void setup() {
  myservo.attach(3);
  radio.begin();
  radio.openReadingPipe(0, address);
  radio.startListening();
}

void loop() {
  if (radio.available()) {
    radio.read(&xValue, sizeof(xValue));

    int angle = map(xValue, 0, 1023, 0, 180);
    myservo.write(angle);
  }
}
