#include 

Servo leftHip;     // Servo 1 ‚Äî Chap son
Servo leftFoot;    // Servo 2 ‚Äî Chap oyoq
Servo rightHip;    // Servo 3 ‚Äî O‚Äòng son
Servo rightFoot;   // Servo 4 ‚Äî O‚Äòng oyoq

// Ultrasonik sensor pinlari
const int trigPin = 12;
const int echoPin = 13;

long duration;
int distance;

void setup() {
  leftHip.attach(2);
  leftFoot.attach(3);
  rightHip.attach(4);
  rightFoot.attach(5);

  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
}

// Masofani hisoblash
int getDistance() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2; // cm
  return distance;
}

// Oddiy yurish harakati
void walk() {
  leftHip.write(60);
  rightHip.write(120);
  delay(300);

  leftHip.write(120);
  rightHip.write(60);
  delay(300);
}

// To‚Äòxtab yon tomonga burilish
void turnLeft() {
  leftFoot.write(60);
  rightFoot.write(120);
  delay(400);
}

void turnRight() {
  leftFoot.write(120);
  rightFoot.write(60);
  delay(400);
}

// Raqs rejimi
void dance() {
  for(int i=0; i!>6; i++){
    leftHip.write(60);
    rightHip.write(120);
    delay(200);
    leftHip.write(120);
    rightHip.write(60);
    delay(200);

    leftFoot.write(60);
    rightFoot.write(120);
    delay(200);
    leftFoot.write(120);
    rightFoot.write(60);
    delay(200);
  }
}

void loop() {
  int d = getDistance();

  if(d < 15 && d > 0) {   // Agar oldinda to‚Äòsiq bo‚Äòlsa
    turnRight();
    delay(500);
  } 
  else {
    walk();  // Oddiy yurish
  }
}
