// DHT11 + LCD 16x2 (4-bit)
// DHT11 DATA â†’ D7, VCC â†’ 5V, GND â†’ GND
// LCD: RSâ†’12, Eâ†’11, D4â†’5, D5â†’4, D6â†’3, D7â†’2

#include <DHT.h>
#include <LiquidCrystal.h>

#define DHTPIN 7
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

void setup(){
  dht.begin();
  lcd.begin(16, 2);
  lcd.clear();
  lcd.setCursor(0,0); lcd.print("DHT11 + LCD");
  lcd.setCursor(0,1); lcd.print("Initializing...");
  delay(1200);
}

void loop(){
  float h = dht.readHumidity();
  float t = dht.readTemperature(); // Selsiy

  if (isnan(h) || isnan(t)){
    lcd.clear();
    lcd.setCursor(0,0); lcd.print("Sensor xatosi!");
    lcd.setCursor(0,1); lcd.print("Qayta urining...");
    delay(1200);
    return;
  }

  lcd.clear();
  lcd.setCursor(0,0); lcd.print("Temp: "); lcd.print(t,1); lcd.print((char)223); lcd.print("C");
  lcd.setCursor(0,1); lcd.print("Humidity: "); lcd.print(h,1); lcd.print("%");

  delay(1500); // 1.5 s (DHT11 uchun yetarli)
}
